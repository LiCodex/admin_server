准备工作需要
	#安装python
	#安装python paramiko工具包


1.更新源
	apt-get update
2.安装mongodb
	apt-get install mongodb
3.安装redis-server
	apt-get install redis-server
4.创建项目文件夹
	cd /home 
	mkdir download
	mkdir proj
5.安装openresty
	cd /home/download
	wget https://openresty.org/download/openresty-1.11.2.5.tar.gz
	tar -zxvf openresty-1.11.2.5.tar.gz

	mv openresty-1.11.2.5 openresty
	cd openresty

	apt-get install libpcre3-dev \
    libssl-dev perl make build-essential curl

    ./configure
    make 
    make install

    touch ~/.bash_profile
    vim ~/.bash_profile
    把这个贴上去
    PATH=/usr/local/openresty/nginx/sbin:$PATH
	export PATH

	保存

	source ~/.bash_profile

6.上传openresty的库
	python uploadrestylib.py

7.打包上传项目的代码
	python upload.py
